function doGet() {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getSheetByName("Banco de Dados");
  var data = sheet.getDataRange().getValues();
  var rows = [];
  for (var i = 1; i < data.length; i++) {
    var row = data[i];
    if (!row[0] || !row[1]) continue;
    var dv = row[0];
    if (dv instanceof Date)
      dv = Utilities.formatDate(dv,"America/Sao_Paulo","yyyy-MM-dd");
    rows.push({
      "Data":dv,"Tipo":String(row[1]||""),
      "Forma de pagamento":String(row[2]||""),
      "Categoria":String(row[3]||""),
      "Descricao":row[4]!==undefined&&row[4]!==""?String(row[4]):"",
      "Valor":row[5]||0
    });
  }
  var saldo=sheet.getRange("J2").getValue();
  var cartao=sheet.getRange("K5").getValue();
  return ContentService
    .createTextOutput(JSON.stringify({data:rows,saldo:saldo,cartao:cartao}))
    .setMimeType(ContentService.MimeType.JSON);
}
